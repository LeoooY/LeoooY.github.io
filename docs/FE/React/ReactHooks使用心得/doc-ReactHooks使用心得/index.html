ReactHooks使用心得

Hook封装子组件#

把子组件封装成一个 useXXX(), 然后 return dom 和需要调用的方法 好处： 父>子之间的通信和调用更方便直接

  • 父组件可以方便的调用子组件方法
  • 父组件可以方便的改变子组件方法
  • 父组件可以方便的获取子组件状态
const useChild = () => {
    const [state, setState] = useState()
    const onChange = useCallback(() => {
    }, [])
    return {
        dom: < child / > ,
        methods: {
            onChange,
            setState,
        }
    }
}

# Problem:
const [deps,setDeps]=useState(0)
const onChange = useCallback(()=>{
    /* 将deps处理为新值 */
    const newDeps=...  // 伪代码，省略
    setDeps(newDeps)
},[deps])
useEffect(()=>{
    onChange()
},[props.do])
/* 问题在于这里，依赖会报错提示将onChange加入依赖 */
/* 但是加入依赖后，useEffect并不能按照设计的props.do改变才触发，而是onChange改变后也会触发 */
# Solution：
const [deps,setDeps]=useState(0)
const onChange = useCallback(()=>{
    /* 将deps处理为新值 */
    const newDeps=...  // 伪代码，省略
    setDeps(newDeps)
},[deps])
/* *********精髓部分******* */
const onChangeRef=useRef(onChange)
useEffect(()=>{
    onChangeRef.current=onChange
},[onChange])
useEffect(()=>{
    onChangeRef.current()
},[props.do])
/* 此时onChangeRef不需要加入依赖 */
/* *********精髓部分********* */

另一种思路

// https://reactjs.org/docs/hooks-faq.html#is-it-safe-to-omit-functions-from-the-list-of-dependencies
function Form() {
    const [text, updateText] = useState('');
    // Will be memoized even if `text` changes:
    const handleSubmit = useEventCallback(() => {
        alert(text);
    }, [text]);
    return ( <
        >
        <
        input value = {
            text
        }
        onChange = {
            e => updateText(e.target.value)
        }
        /> <
        ExpensiveTree onSubmit = {
            handleSubmit
        }
        /> < / >
    );
}
function useEventCallback(fn, dependencies) {
    const ref = useRef(() => {
        throw new Error('Cannot call an event handler while rendering.');
    });
    useEffect(() => {
        ref.current = fn;
    }, [fn, ...dependencies]);
    return useCallback(() => {
        const fn = ref.current;
        return fn();
    }, [ref]);
}
