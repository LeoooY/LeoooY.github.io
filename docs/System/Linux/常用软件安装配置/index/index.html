git#

  • 安装
yum install git
sudo apt install git
  • 配置
git config --global user.name "yourname" #引号里面输入你的名字2
git config --global user.email "youremail" #输入邮箱
git config --global core.autocrlf false #消除由于Windows和Linux平台中换行符的差异导致的问题
git config --global core.quotepath off #消除由于路径或者是文件名包含中文导致的乱码问题
git config --global gui.encoding utf-8 #消除gui界面中文乱码问题(如果全程使用命令行的话不用担心这个问题)
ssh-keygen -t rsa -C "youremail" #配置ssh的密钥，输完之后一路回车
eval `ssh-agent` #启用ssh-agent
ssh-add ~/.ssh/id_rsa #添加密钥
ssh-add -l #将它添加到已知的key列表中
cat ~/.ssh/id_rsa.pub #把这个公钥添加到自己的Github账户上去
git config --global credential.helper store  //存储密码

nodeJs#

get nodeJs and installl

# Using Ubuntu
curl -sL https://deb.nodesource.com/setup_12.x | sudo -E bash -
sudo apt-get install -y nodejs
  • 下载
curl -O https://nodejs.org/dist/v12.13.0/node-v12.13.0-linux-x64.tar.xz
// 官网找binary for linux，推荐使用LTS版本，稳定
// 注意不要下载成源码了，源码的话需要自己编译makefile
  • 解压
tar -jxv -f nodejsxxxxx.tar.xz
tar -xv -f nodejsxxxxx.tar.xz
  • 配置环境变量vim ~/.profile or vim ~/.bashrc
    文件末尾插入内容：
# Nodejs
VERSION=v12.13.0
DISTRO=linux-x64
export PATH=/usr/LY/download/nodeJs/node-$VERSION-$DISTRO/bin:$PATH

版本填写对应下载版本 路径为nodeJs解压后的路径/bin

  • 安装完成，测试
node -v
npm version
npx -v
  • 创建软连接 why create soft link?
sudo ln -s /usr/local/lib/nodejs/node-$VERSION-$DISTRO/bin/node /usr/bin/node
sudo ln -s /usr/local/lib/nodejs/node-$VERSION-$DISTRO/bin/npm /usr/bin/npm
sudo ln -s /usr/local/lib/nodejs/node-$VERSION-$DISTRO/bin/npx /usr/bin/npx
  • 配置npm
npm config set cache=/usr/LY/data/nodeJs/node_cache
npm config set prefix=/usr/LY/data/nodeJs/node_global

Mongo#

  • 编辑mongo yum源
vim /etc/yum.repos.d/mongodb.repo
//content
[MongoDB]
name=MongoDB Repository
baseurl=http://repo.mongodb.org/yum/redhat/$releasever/mongodb-org/4.2/$basearch/
gpgcheck=1
enabled=1
gpgkey=https://www.mongodb.org/static/pgp/server-4.2.asc
sudo yum install mongodb-org
systemctl start mongod.service    # For CentOS 8/7
//Configure MongoDB to autostart on system boot
systemctl enable mongod.service    # For CentOS 8/7

ElasticSearch#

  • install Java before start

  • Setup Yum Repository

// install GPG key for the elasticsearch rpm packages.
sudo rpm --import https://artifacts.elastic.co/GPG-KEY-elasticsearch
// create yum repository file for the elasticsearch.
sudo vi /etc/yum.repos.d/elasticsearch.repo
// content:
[Elasticsearch-7]
name=Elasticsearch repository for 7.x packages
baseurl=https://artifacts.elastic.co/packages/7.x/yum
gpgcheck=1
gpgkey=https://artifacts.elastic.co/GPG-KEY-elasticsearch
enabled=1
autorefresh=1
type=rpm-md
  • Install Elasticsearch
sudo yum install elasticsearch
  • Setup Elasticsearch
vim /etc/elasticsearch/elasticsearch.yml
// content:
network.host: localhost
  • Start Service
sudo systemctl enable elasticsearch
sudo systemctl start elasticsearch
curl -X<VERB> '<PROTOCOL>://<HOST>:<PORT>/<PATH>?<QUERY_STRING>' -d '<BODY>'
curl -X GET localhost:9200
